26 static VKAPI_ATTR VkBool32 VKAPI_CALL debugReportCallback(
27  VkDebugReportFlagsEXT flags,
28  VkDebugReportObjectTypeEXT /*objType*/,
29  uint64_t /*obj*/,
30  size_t /*location*/,
31  int32_t /*code*/,
32  const char* layerPrefix,
33  const char* msg,
34  void* /*userData*/) {
35 
36  // Convert VkDebugReportFlagsEXT to System::Logger::Level
37  System::Logger::Level level = System::Logger::Level::Off;
38 
39  if (flags & VK_DEBUG_REPORT_ERROR_BIT_EXT) {
40  level = System::Logger::Level::Error;
41  } else if (flags & VK_DEBUG_REPORT_WARNING_BIT_EXT || flags & VK_DEBUG_REPORT_PERFORMANCE_WARNING_BIT_EXT) {
42  level = System::Logger::Level::Warning;
43  } else if (flags & VK_DEBUG_REPORT_INFORMATION_BIT_EXT) {
44  level = System::Logger::Level::Info;
45  } else if (flags & VK_DEBUG_REPORT_DEBUG_BIT_EXT) {
46  level = System::Logger::Level::Debug;
47  }
48 
49  LUG_LOG.log(level, "{}: {}", layerPrefix, msg);
50 
51  return VK_FALSE;
52 }

157  }
158  }
159  }
160

163  // Check which layers / extensions to load for modules

197  return false;

201  }
202

204  {
205  if (isInstanceExtensionLoaded(VK_EXT_DEBUG_REPORT_EXTENSION_NAME)) {
206  VkDebugReportCallbackCreateInfoEXT createInfo = {};

280  }
281

286  LUG_VULKAN_FORMAT(LUG_LOAD_IMAGE_FORMAT_PROPERTIES)

297 bool Renderer::initDevice() {
298  VkResult result;
299 
300  // Select device
301  {
302  uint8_t matchedDeviceIdx = 0;
303  std::vector<uint8_t> matchedDevicesIdx{};
304 
305  for (uint8_t idx = 0; idx < _physicalDeviceInfos.size(); ++idx) {
306  if (!checkRequirementsDevice(_physicalDeviceInfos[idx], _graphics.getLoadedMandatoryModules(), false)) {
307  continue;
308  }
309 
310  checkRequirementsDevice(_physicalDeviceInfos[idx], _graphics.getLoadedOptionalModules(), false);
311 
312  matchedDevicesIdx.push_back(idx);
313 
314  }
315 
316  if (matchedDevicesIdx.size() == 0) {
317  LUG_LOG.error("RendererVulkan: Can't find a compatible device");
318  return false;
319  }
320 
321  // TODO: Add score
322  matchedDeviceIdx = matchedDevicesIdx[0];
323  for (uint8_t idx : matchedDevicesIdx) {
324  if (_physicalDeviceInfos[idx].properties.deviceType == ((_initInfo.useDiscreteGPU == true) ? (VK_PHYSICAL_DEVICE_TYPE_DISCRETE_GPU) : (VK_PHYSICAL_DEVICE_TYPE_INTEGRATED_GPU))) {
325  matchedDeviceIdx = idx;
326  break;
327  }
328  }
329 
330  _physicalDeviceInfo = &(_physicalDeviceInfos[matchedDeviceIdx]);
331  }
332 
333  // Set the loaded informations of the matched device
334  {
335  checkRequirementsDevice(*_physicalDeviceInfo, _graphics.getLoadedMandatoryModules(), true);
336  checkRequirementsDevice(*_physicalDeviceInfo, _graphics.getLoadedOptionalModules(), true);
337  }
338 
339  // Create device
340  {
341  float queuePriority = 1.0f;
342  std::vector<VkDeviceQueueCreateInfo> queueCreateInfo(_loadedQueueFamiliesIdx.size());
343 
344  {
345  uint8_t i = 0;
346 
347  for (auto idx : _loadedQueueFamiliesIdx) {
348  queueCreateInfo[i].sType = VK_STRUCTURE_TYPE_DEVICE_QUEUE_CREATE_INFO;
349  queueCreateInfo[i].queueFamilyIndex = idx;
350  queueCreateInfo[i].queueCount = 1;
351  queueCreateInfo[i].pQueuePriorities = &queuePriority;
352 
353  ++i;
354  }
355  }
356 
357  VkDeviceCreateInfo createInfo{
358  createInfo.sType = VK_STRUCTURE_TYPE_DEVICE_CREATE_INFO,
359  createInfo.pNext = nullptr,
360  createInfo.flags = 0,
361  createInfo.queueCreateInfoCount = static_cast<uint32_t>(queueCreateInfo.size()),
362  createInfo.pQueueCreateInfos = queueCreateInfo.data(),
363  createInfo.enabledLayerCount = 0, // Deprecated
364  createInfo.ppEnabledLayerNames = nullptr, // Deprecated
365  createInfo.enabledExtensionCount = static_cast<uint32_t>(_loadedDeviceExtensions.size()),
366  createInfo.ppEnabledExtensionNames = _loadedDeviceExtensions.data(),
367  createInfo.pEnabledFeatures = &_loadedDeviceFeatures
368  };
369 
370  VkDevice device{VK_NULL_HANDLE};
371  result = vkCreateDevice(_physicalDeviceInfo->handle, &createInfo, nullptr, &device);
372  if (result != VK_SUCCESS) {
373  LUG_LOG.error("RendererVulkan: Can't create the device: {}", result);
374  return false;
375  }
376 
377  _device = API::Device(device, _physicalDeviceInfo);
378  }
379 
380  // Load vulkan device functions
381  {
382  if (!_loader.loadDeviceFunctions(_device)) {
383  LUG_LOG.error("RendererVulkan: Can't load device vulkan functions");
384  return false;
385  }
386  }
387 
388  // Create queues
389  {
390  _queues.resize(_loadedQueueFamiliesIdx.size());
391 
392  uint8_t i = 0;
393  for (auto idx : _loadedQueueFamiliesIdx) {
394  VkQueue queue;
395  vkGetDeviceQueue(static_cast<VkDevice>(_device), idx, 0, &queue);
396 
397  _queues[i] = API::Queue(idx, queue, _physicalDeviceInfo->queueFamilies[idx].queueFlags);
398 
399  ++i;
400  }
401  }
402 
403  // Create command pools
404  {
405  for (auto& queue : _queues) {
406  VkCommandPoolCreateInfo createInfo{
407  createInfo.sType = VK_STRUCTURE_TYPE_COMMAND_POOL_CREATE_INFO,
408  createInfo.pNext = nullptr,
409  createInfo.flags = VK_COMMAND_POOL_CREATE_RESET_COMMAND_BUFFER_BIT, // TODO: Maybe change that
410  createInfo.queueFamilyIndex = queue.getFamilyIdx()
411  };
412 
413  VkCommandPool commandPool{VK_NULL_HANDLE};
414  result = vkCreateCommandPool(static_cast<VkDevice>(_device), &createInfo, nullptr, &commandPool);
415  if (result != VK_SUCCESS) {
416  LUG_LOG.error("RendererVulkan: Can't create a command pool: {}", result);
417  return false;
418  }
419 
420  queue.setCommandPool(API::CommandPool(commandPool, &_device, &queue));
421  }
422  }
423  return true;
424 }

426 bool Renderer::checkRequirementsInstance(const std::set<Module::Type> &modulesToCheck) {
427  bool requirementsCheck = true;
428 
429  for (const auto moduleType : modulesToCheck) {
430  const auto& requirements = modulesRequirements.at(moduleType);
431 
432  std::vector<const char*> layers{};
433  std::vector<const char*> extensions{};
434 
435  bool moduleRequirementsCheck = true;
436 
437  // Check layers
438  {
439  const std::vector<const char*> layersNotFound = checkRequirementsLayers(_instanceInfo, requirements.mandatoryInstanceLayers, layers);
440  moduleRequirementsCheck = moduleRequirementsCheck && layersNotFound.size() == 0;
441 
442  for (const char* const layerName : layersNotFound) {
443  LUG_LOG.warn("Can't load mandatory layer '{}' for module '{}'", layerName, moduleType);
444  }
445  }
446 
447  {
448  const std::vector<const char*> layersNotFound = checkRequirementsLayers(_instanceInfo, requirements.optionalInstanceLayers, layers);
449 
450  for (const char* layerName : layersNotFound) {
451  LUG_LOG.warn("Can't load optional layer '{}' for module '{}'", layerName, moduleType);
452  }
453  }
454 
455  // Check extensions
456  {
457  const std::vector<const char*> extensionsNotFound = checkRequirementsExtensions(_instanceInfo, requirements.mandatoryInstanceExtensions, extensions);
458  moduleRequirementsCheck = moduleRequirementsCheck && extensionsNotFound.size() == 0;
459 
460  for (const char* const extensionName : extensionsNotFound) {
461  LUG_LOG.warn("Can't load mandatory extension '{}' for module '{}'", extensionName, moduleType);
462  }
463  }
464 
465  {
466  const std::vector<const char*> extensionsNotFound = checkRequirementsExtensions(_instanceInfo, requirements.optionalInstanceExtensions, extensions);
467 
468  for (const char* extensionName : extensionsNotFound) {
469  LUG_LOG.warn("Can't load optional extension '{}' for module '{}'", extensionName, moduleType);
470  }
471  }
472 
473  if (moduleRequirementsCheck) {
474  _loadedInstanceLayers.insert(_loadedInstanceLayers.end(), layers.begin(), layers.end());
475  _loadedInstanceExtensions.insert(_loadedInstanceExtensions.end(), extensions.begin(), extensions.end());
476  } else {
477  _graphics.unsupportedModule(moduleType);
478  }
479 
480  requirementsCheck = requirementsCheck && moduleRequirementsCheck;
481  }
482 
483  return requirementsCheck;
484 }

486 bool Renderer::checkRequirementsDevice(const PhysicalDeviceInfo& physicalDeviceInfo, const std::set<Module::Type> &modulesToCheck, bool finalization) {
487  bool requirementsCheck = true;
488 
489  for (const auto moduleType : modulesToCheck) {
490  const auto& requirements = modulesRequirements.at(moduleType);
491 
492  std::vector<const char*> extensions{};
493  VkPhysicalDeviceFeatures features{};
494  std::set<int8_t> queueFamiliesIdx{};
495 
496  bool moduleRequirementsCheck = true;
497 
498  // Check extensions
499  {
500  const std::vector<const char*> extensionsNotFound = checkRequirementsExtensions(physicalDeviceInfo, requirements.mandatoryDeviceExtensions, extensions);
501  moduleRequirementsCheck = moduleRequirementsCheck && extensionsNotFound.size() == 0;
502 
503  if (!finalization) {
504  for (const char* const extensionName : extensionsNotFound) {
505  LUG_LOG.warn("Device {}: Can't load mandatory extension '{}' for module '{}'", physicalDeviceInfo.properties.deviceName, extensionName, moduleType);
506  }
507  }
508  }
509 
510  {
511  const std::vector<const char*> extensionsNotFound = checkRequirementsExtensions(physicalDeviceInfo, requirements.optionalDeviceExtensions, extensions);
512 
513  if (!finalization) {
514  for (const char* extensionName : extensionsNotFound) {
515  LUG_LOG.warn("Device {}: Can't load optional extension '{}' for module '{}'", physicalDeviceInfo.properties.deviceName, extensionName, moduleType);
516  }
517  }
518  }
519 
520  // TODO: Log error
521  #define LUG_CHECK_VULKAN_PHYSICAL_DEVICE_MANDATORY_FEATURES(featureName) \
522  { \
523  if (requirements.mandatoryFeatures.featureName == VK_TRUE) { \
524  if (physicalDeviceInfo.features.featureName == VK_TRUE) { \
525  features.featureName = VK_TRUE; \
526  } else { \
527  if (!finalization) { \
528  LUG_LOG.warn("Device {}: Can't load mandatory feature '{}' for module '{}'", physicalDeviceInfo.properties.deviceName, #featureName, moduleType); \
529  } \
530  \
531  moduleRequirementsCheck = false; \
532  } \
533  } \
534  }
535  LUG_VULKAN_PHYSICAL_DEVICE_FEATURES(LUG_CHECK_VULKAN_PHYSICAL_DEVICE_MANDATORY_FEATURES);
536  #undef LUG_CHECK_VULKAN_PHYSICAL_DEVICE_MANDATORY_FEATURES
537 
538  // TODO: Log error
539  #define LUG_CHECK_VULKAN_PHYSICAL_DEVICE_OPTIONNAL_FEATURES(featureName) \
540  { \
541  if (requirements.optionalFeatures.featureName == VK_TRUE) { \
542  if (physicalDeviceInfo.features.featureName == VK_TRUE) { \
543  features.featureName = VK_TRUE; \
544  } else if (!finalization) { \
545  LUG_LOG.warn("Device {}: Can't load optional feature '{}' for module '{}'", physicalDeviceInfo.properties.deviceName, #featureName, moduleType); \
546  } \
547  } \
548  }
549  LUG_VULKAN_PHYSICAL_DEVICE_FEATURES(LUG_CHECK_VULKAN_PHYSICAL_DEVICE_OPTIONNAL_FEATURES);
550  #undef LUG_CHECK_VULKAN_PHYSICAL_DEVICE_MANDATORY_FEATURES
551 
552  for (const auto& queueFlags : requirements.mandatoryQueueFlags) {
553  int8_t idx = 0;
554  if (physicalDeviceInfo.containsQueueFlags(queueFlags, idx)) {
555  queueFamiliesIdx.insert(idx);
556  } else {
557  if (!finalization) {
558  LUG_LOG.warn("Device {}: Can't find mandatory queue type for module '{}'", physicalDeviceInfo.properties.deviceName, moduleType);
559  }
560 
561  moduleRequirementsCheck = false;
562  }
563  }
564 
565  for (const auto& queueFlags : requirements.optionalQueueFlags) {
566  int8_t idx = 0;
567  if (physicalDeviceInfo.containsQueueFlags(queueFlags, idx)) {
568  queueFamiliesIdx.insert(idx);
569  } else if (!finalization) {
570  LUG_LOG.warn("Device {}: Can't find optional queue type for module '{}'", physicalDeviceInfo.properties.deviceName, moduleType);
571  }
572  }
573 
574  if (finalization) {
575  if (moduleRequirementsCheck) {
576  _loadedDeviceExtensions.insert(_loadedDeviceExtensions.end(), extensions.begin(), extensions.end());
577 
578  #define LUG_FILL_VULKAN_PHYSICAL_DEVICE_FEATURES(featureName) _loadedDeviceFeatures.featureName = _loadedDeviceFeatures.featureName || features.featureName;
579  LUG_VULKAN_PHYSICAL_DEVICE_FEATURES(LUG_FILL_VULKAN_PHYSICAL_DEVICE_FEATURES);
580  #undef LUG_FILL_VULKAN_PHYSICAL_DEVICE_FEATURES
581 
582  _loadedQueueFamiliesIdx.insert(queueFamiliesIdx.begin(), queueFamiliesIdx.end());
583  } else {
584  _graphics.unsupportedModule(moduleType);
585  }
586  }
587 
588  requirementsCheck = requirementsCheck && moduleRequirementsCheck;
589  }
590 
591  return requirementsCheck;
