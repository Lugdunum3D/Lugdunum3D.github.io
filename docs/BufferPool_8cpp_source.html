25 BufferPool::SubBuffer* BufferPool::allocate() {
26  // Search free sub-buffer
27  {
28  for (auto& chunk: _chunks) {
29  BufferPool::SubBuffer* subBuffer = chunk->getFreeSubBuffer();
30  if (subBuffer) {
31  return subBuffer;
32  }
33  }
34  }
35 
36  // No free sub-buffer found, allocate new chunk
37  uint32_t subBufferSizeAligned = API::Buffer::getSizeAligned(_device, _subBufferSize);
38  std::unique_ptr<BufferPool::Chunk> chunk = std::make_unique<BufferPool::Chunk>();
39  {
40  // Create buffer
41  chunk->size = subBufferSizeAligned * _countPerChunk;
42  chunk->buffer = API::Buffer::create(_device,
43  (uint32_t)_queueFamilyIndices.size(),
44  _queueFamilyIndices.data(),
45  chunk->size,
46  VK_BUFFER_USAGE_UNIFORM_BUFFER_BIT | VK_BUFFER_USAGE_TRANSFER_DST_BIT);
47  if (!chunk->buffer) {
48  return nullptr;
49  }
50 
51  // Allocate memory for buffer
52  const VkMemoryRequirements& bufferRequirements = chunk->buffer->getRequirements();
53  uint32_t memoryTypeIndex = API::DeviceMemory::findMemoryType(_device, bufferRequirements, VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
54  chunk->bufferMemory = API::DeviceMemory::allocate(_device, chunk->size, memoryTypeIndex);
55  if (!chunk->bufferMemory) {
56  return nullptr;
57  }
58 
59  // Bind memory to buffer
60  chunk->buffer->bindMemory(chunk->bufferMemory.get(), 0);
61 
62  // Create and update descriptor set
63  std::vector<API::DescriptorSet> descriptorSets = _descriptorPool->createDescriptorSets({static_cast<VkDescriptorSetLayout>(*_descriptorSetLayout)});
64  if (descriptorSets.size() == 0) {
65  return nullptr;
66  }
67  chunk->descriptorSet = std::move(descriptorSets[0]);
68  chunk->descriptorSet.update(VK_DESCRIPTOR_TYPE_UNIFORM_BUFFER_DYNAMIC , 0, chunk->buffer.get(), 0, subBufferSizeAligned);
69  }
70 
71  // Init new chunk sub-buffers
72  {
73  chunk->subBuffers.resize(_countPerChunk);
74  chunk->subBuffersFree.resize(_countPerChunk);
75  uint32_t offset = 0;
76  for (uint32_t i = 0; i < _countPerChunk; ++i) {
77  SubBuffer subBuffer{&chunk->descriptorSet,
78  chunk->buffer.get(),
79  offset,
80  subBufferSizeAligned,
81  chunk.get()
82  };
83  chunk->subBuffers[i] = std::move(subBuffer);
84 
85  // Warning: Do not resize chunk->subBuffers
86  chunk->subBuffersFree[i] = &chunk->subBuffers[i];
87 
88  offset += subBufferSizeAligned;
89  }
90  }
91 
92  _chunks.push_back(std::move(chunk));
93 
94  return allocate();
95 }
