335  VkResult result;
336  PhysicalDeviceInfo* info = _renderer.getPhysicalDeviceInfo();
337 
338  LUG_ASSERT(info != nullptr, "PhysicalDeviceInfo cannot be null");
339 
340  // TODO: Find a way to put Preferences elsewhere
341  Renderer::Preferences::Swapchain& swapchainPreferences = _renderer.getPreferences().swapchain;
342 
343  VkPresentModeKHR swapchainPresentMode = VK_PRESENT_MODE_MAX_ENUM_KHR;
344  VkSurfaceFormatKHR swapchainFormat{
345  swapchainFormat.format = VK_FORMAT_MAX_ENUM,
346  swapchainFormat.colorSpace = VK_COLOR_SPACE_SRGB_NONLINEAR_KHR
347  };
348  VkCompositeAlphaFlagBitsKHR compositeAlpha = static_cast<VkCompositeAlphaFlagBitsKHR>(0);
349  uint32_t minImageCount = 3;
350  VkExtent2D extent{};
351  VkSurfaceTransformFlagsKHR transform{};
352 
353  // Check requirements for swapchain
354  {
355  // Check the present mode
356  {
357  for (auto presentMode : swapchainPreferences.presentModes) {
358  if (std::find(info->swapchain.presentModes.begin(), info->swapchain.presentModes.end(), presentMode) != info->swapchain.presentModes.end()) {
359  LUG_LOG.info("RendererWindow: Use present mode {}", API::RTTI::toStr(presentMode));
360 
361  swapchainPresentMode = presentMode;
362  break;
363  }
364  }
365 
366  if (swapchainPresentMode == VK_PRESENT_MODE_MAX_ENUM_KHR) {
367  LUG_LOG.error("RendererWindow: Missing present mode supported by Lugdunum");
368  return false;
369  }
370  }
371 
372  // Check the formats
373  {
374  for (auto format : swapchainPreferences.formats) {
375  if (std::find_if(info->swapchain.formats.begin(), info->swapchain.formats.end(), [&swapchainFormat, &format](const VkSurfaceFormatKHR& lhs) {
376  return lhs.colorSpace == swapchainFormat.colorSpace && format == lhs.format;
377  }) != info->swapchain.formats.end()) {
378  LUG_LOG.info("RendererWindow: Use format {}", API::RTTI::toStr(format));
379 
380  swapchainFormat.format = format;
381  break;
382  }
383  }
384 
385  if (swapchainFormat.format == VK_FORMAT_MAX_ENUM) {
386  LUG_LOG.error("RendererWindow: Missing swapchain format supported by Lugdunum");
387  return false;
388  }
389  }
390 
391  // Check composite alpha
392  {
393  for (auto compositeAlphaPreferency : swapchainPreferences.compositeAlphas) {
394  if (info->swapchain.capabilities.supportedCompositeAlpha & compositeAlphaPreferency) {
395  LUG_LOG.info("RendererWindow: Use composite alpha {}", API::RTTI::toStr(compositeAlphaPreferency));
396 
397  compositeAlpha = compositeAlphaPreferency;
398  break;
399  }
400  }
401 
402  if (!compositeAlpha) {
403  LUG_LOG.error("RendererWindow: Missing composite alpha supported by Lugdunum");
404  return false;
405  }
406  }
407 
408  // Check image counts
409  if (info->swapchain.capabilities.maxImageCount > 0 && info->swapchain.capabilities.maxImageCount < minImageCount) {
410  LUG_LOG.error("RendererWindow: Not enough images ({} required), found {}", minImageCount, info->swapchain.capabilities.maxImageCount);
411  return false;
412  }
413 
414  // If width (and height) equals the special value 0xFFFFFFFF, the size of the surface will be set by the swapchain
415  if (info->swapchain.capabilities.currentExtent.height == 0xFFFFFFFF
416  && info->swapchain.capabilities.currentExtent.width == 0xFFFFFFFF) {
417  extent.height = _mode.height;
418  extent.width = _mode.width;
419  } else {
420  extent.height = info->swapchain.capabilities.currentExtent.height;
421  extent.width = info->swapchain.capabilities.currentExtent.width;
422  }
423

425  if (info->swapchain.capabilities.supportedTransforms & VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR) {
426  // We prefer a non-rotated transform
427  transform = VK_SURFACE_TRANSFORM_IDENTITY_BIT_KHR;
428  } else {
429  transform = info->swapchain.capabilities.currentTransform;
430  }
431  }
432

434  {

437  createInfo.pNext = nullptr,
438  createInfo.flags = 0,
439  createInfo.surface = _surface,
440  createInfo.minImageCount = minImageCount,
441  createInfo.imageFormat = swapchainFormat.format,
442  createInfo.imageColorSpace = swapchainFormat.colorSpace,
443  {}, // createInfo.imageExtent
444  createInfo.imageArrayLayers = 1,
445  createInfo.imageUsage = VK_IMAGE_USAGE_COLOR_ATTACHMENT_BIT,
446  createInfo.imageSharingMode = VK_SHARING_MODE_EXCLUSIVE,
