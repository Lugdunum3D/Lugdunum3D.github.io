374 void WindowImpl::processWindowEvents(UINT message, WPARAM wParam, LPARAM lParam) {
375  Event e;
376 
377  switch (message) {
378  case WM_SIZE:
379  e.type = Event::Type::Resize;
380 
381  // The handle is not set on first call
382  if (!_handle) {
383  break;
384  }
385 
386  RECT rect;
387 
388  if (GetClientRect(_handle, &rect)) {
389  _parent->_mode.width = static_cast<uint16_t>(rect.right - rect.left);
390  _parent->_mode.height = static_cast<uint16_t>(rect.bottom - rect.top);
391  } else {
392  LUG_LOG.error("WindowImpl::Win32::processWindowEvents Can't get window rect");
393  }
394  break;
395 
396  case WM_CLOSE:
397  e.type = Event::Type::Close;
398  break;
399 
400  case WM_DESTROY:
401  e.type = Event::Type::Destroy;
402  break;
403 
404  case WM_KEYDOWN:
405  case WM_SYSKEYDOWN:
406  if (_keyRepeat || ((HIWORD(lParam) & KF_REPEAT) == 0)) {
407  e.type = Event::Type::KeyPressed;
408  configKeyEvent(e.key, wParam, lParam);
409  } else {
410  return;
411  }
412  break;
413 
414  case WM_KEYUP:
415  case WM_SYSKEYUP:
416  e.type = Event::Type::KeyReleased;
417  configKeyEvent(e.key, wParam, lParam);
418  break;
419 
420  case WM_CHAR:
421  if (_keyRepeat || ((lParam & (1 << 30)) == 0)) {
422  e.type = Event::Type::CharEntered;
423  e.character.val = static_cast<wchar_t>(wParam);
424  } else {
425  return;
426  }
427  break;
428 
429  case WM_RBUTTONDOWN:
430  case WM_LBUTTONDOWN:
431  case WM_MBUTTONDOWN:
432  case WM_XBUTTONDOWN:
433  e.type = Event::Type::ButtonPressed;
434  configMouseEvent(e.mouse, wParam, lParam);
435  configMouseButtonEvent(e.mouse, message, wParam, lParam);
436  break;
437 
438  case WM_RBUTTONUP:
439  case WM_LBUTTONUP:
440  case WM_MBUTTONUP:
441  case WM_XBUTTONUP:
442  e.type = Event::Type::ButtonReleased;
443  configMouseEvent(e.mouse, wParam, lParam);
444  configMouseButtonEvent(e.mouse, message, wParam, lParam);
445  break;
446 
447  case WM_MOUSEMOVE:
448  e.type = Event::Type::MouseMoved;
449  e.mouse.code = Mouse::Button::Unknown;
450  configMouseEvent(e.mouse, wParam, lParam);
451  break;
452 
453  case WM_MOUSEWHEEL:
454  case WM_MOUSEHWHEEL:
455  e.type = Event::Type::MouseWheel;
456  configMouseEvent(e.mouse, wParam, lParam);
457  configMouseWheelEvent(e.mouse, message, wParam);
458  break;
459 
460  default:
461  return;
462  }
463 
464  _events.push(std::move(e));
465 }

489

491  devMode.dmPelsWidth = _parent->_mode.width; // Selected Screen Width
492  devMode.dmPelsHeight = _parent->_mode.height; // Selected Screen Height
493  devMode.dmBitsPerPel = 32; // Selected Bits Per Pixel
494  devMode.dmFields = DM_BITSPERPEL | DM_PELSWIDTH | DM_PELSHEIGHT;
495 
496  // Apply fullscreen mode

498  // If The Mode Fails, Offer Two Options. Quit Or Use Windowed Mode.
499  if (MessageBox(NULL, "Fullscreen mode has failed to be initialized. Use Windowed Mode Instead?", "Warning", MB_YESNO | MB_ICONEXCLAMATION) == IDYES) {
500  _fullscreen = false;
501  return false;
502  } else {
503  // Pop Up A Message Box Letting User Know The Program Is Closing.
504  MessageBox(NULL, "Program Will Now Close.", "ERROR", MB_OK | MB_ICONSTOP);
505  return false;
506  }
507  }
508 
509  // Make the window flags compatible with fullscreen mode
510  SetWindowLongW(_handle, GWL_STYLE, WS_POPUP | WS_CLIPCHILDREN | WS_CLIPSIBLINGS);
511  SetWindowLongW(_handle, GWL_EXSTYLE, WS_EX_APPWINDOW);
512 
513  // Resize the window so that it fits the entire screen

515  ShowWindow(_handle, SW_SHOW);
