330 bool Forward::initDepthBuffers(const std::vector<std::unique_ptr<API::ImageView>>& imageViews) {
331  VkFormat imagesFormat = API::Image::findSupportedFormat(
332  _device,
333  {VK_FORMAT_D32_SFLOAT, VK_FORMAT_D32_SFLOAT_S8_UINT, VK_FORMAT_D24_UNORM_S8_UINT},
334  VK_IMAGE_TILING_OPTIMAL,
335  VK_FORMAT_FEATURE_DEPTH_STENCIL_ATTACHMENT_BIT);
336 
337  if (imagesFormat == VK_FORMAT_UNDEFINED) {
338  LUG_LOG.error("Forward: Can't find supported format for depth buffer");
339  return false;
340  }
341 
342  if (_depthBufferMemory) {
343  _depthBufferMemory.reset();
344  }
345 
346  _framesData.resize(imageViews.size());
347 
348  for (uint32_t i = 0; i < imageViews.size(); ++i) {
349  std::unique_ptr<API::Image> image = nullptr;
350  std::unique_ptr<API::ImageView> imageView = nullptr;
351 
352  VkExtent3D extent{
353  extent.width = imageViews[i]->getExtent().width,
354  extent.height = imageViews[i]->getExtent().height,
355  extent.depth = 1
356  };
357 
358  // Create depth buffer image
359  {
360  image = API::Image::create(_device, imagesFormat, extent, VK_IMAGE_USAGE_DEPTH_STENCIL_ATTACHMENT_BIT);
361 
362  if (!image) {
363  LUG_LOG.error("Forward: Can't create depth buffer image");
364  return false;
365  }
366 
367  auto& imageRequirements = image->getRequirements();
368 
369  size_t realSize = imageRequirements.size;
370  if (realSize % imageRequirements.alignment) {
371  realSize += imageRequirements.alignment - realSize % imageRequirements.alignment;
372  }
373 
374  // Initialize depth buffer memory (This memory is common for all depth buffer images)
375  if (!_depthBufferMemory) {
376  uint32_t memoryTypeIndex = API::DeviceMemory::findMemoryType(_device, imageRequirements, VK_MEMORY_PROPERTY_DEVICE_LOCAL_BIT);
377  // Allocate image requirements size for all images
378  _depthBufferMemory = API::DeviceMemory::allocate(_device, realSize * imageViews.size(), memoryTypeIndex);
379 
380  if (!_depthBufferMemory) {
381  LUG_LOG.error("Forward: Can't allocate device memory for depth buffer images");
382  return false;
383  }
384  }
385 
386  // Bind memory to image
387  image->bindMemory(_depthBufferMemory.get(), realSize * i);
388  }
389 
390  // Create depth buffer image view
391  {
392  imageView = API::ImageView::create(_device, image.get(), imagesFormat, VK_IMAGE_ASPECT_DEPTH_BIT);
393 
394  if (!imageView) {
395  LUG_LOG.error("Forward: Can't create depth buffer image view");
396  return false;
397  }
398  }
399 
400  _framesData[i].depthBuffer.image = std::move(image);
401  _framesData[i].depthBuffer.imageView = std::move(imageView);
402  }
403 
404  return true;
